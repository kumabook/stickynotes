488 Sticky.prototype.drag = function(elem, e){
489     var that = this;
490     var URL = window.content.document.location.href;
491     var startX = e.clientX, startY = e.clientY;
492     var origX = elem.offsetLeft, origY = elem.offsetTop;
493     var deltaX = startX - origX, deltaY = startY - origY;
494     document.addEventListener("mousemove", moveHandler, true);
495     document.addEventListener("mouseup", upHandler, true);
496     
497     e.stopPropagation();
498     e.preventDefault();
499 
500     function moveHandler(e) {
501         elem.style.left = (e.clientX - deltaX) + "px";
502         elem.style.top = (e.clientY - deltaY) + "px";
503         e.stopPropagation();
504     }
505     
506     function upHandler(e) {
507         document.removeEventListener("mouseup", upHandler, true);
508         document.removeEventListener("mousemove", moveHandler, true);
509 	
510         that.left = parseInt(elem.style.left);//
511         that.top = parseInt(elem.style.top);
512         that.update();
513         e.stopPropagation();
514     }
